.fb_reset {

    background: none;            /* 背景设为无 */

    border: 0;                   /* 边框设为无 */

    border-spacing: 0;           /* 边框间距设为0 */

    color: #000;                 /* 文字颜色为黑色 */

    cursor: auto;                /* 鼠标指针样式为默认 */

    direction: ltr;              /* 文本方向为从左到右 */

    font-family: 'lucida grande', tahoma, verdana, arial, sans-serif; /* 字体系列，以及备选字体 */

    font-size: 12px;             /* 字体大小为12像素 */

    font-style: normal;          /* 字体样式为正常 */

    font-variant: normal;        /* 字体变体为正常 */

    font-weight: normal;         /* 字体粗细为正常 */

    letter-spacing: normal;      /* 字母间距为正常 */

    line-height: 1;              /* 行高为1，通常表示正常的行高 */

    margin: 0;                   /* 外边距设为0 */

    overflow: visible;           /* 内容溢出时显示 */

    padding: 0;                  /* 内边距设为0 */

    text-align: left;            /* 文本对齐方式为左对齐 */

    text-decoration: none;       /* 文本装饰设为无 */

    text-indent: 0;              /* 首行缩进设为0 */

    text-shadow: none;           /* 文本阴影设为无 */

    text-transform: none;        /* 文本转换设为无 */

    visibility: visible;         /* 元素可见 */

    white-space: normal;         /* 空白处理为正常 */

    word-spacing: normal;        /* 单词间距为正常 */

}


.fb_reset {

background: none;：移除背景颜色或图片。

 

border: 0;：移除边框。

 

border-spacing: 0;：移除表格的单元格间隙。

 

color: #000;：设置文本颜色为黑色。

 

cursor: auto;：使用默认光标。

 

direction: ltr;：设置文本方向为从左到右（Left-to-Right）。

 

font-family: 'lucida grande', tahoma, verdana, arial, sans-serif;：设置默认字体。

 

font-size: 12px;：设置默认字体大小为 12 像素。

 

font-style: normal;：设置字体样式为正常。

 

font-variant: normal;：设置字体变体为正常。

 

font-weight: normal;：设置字体粗细为正常。

 

letter-spacing: normal;：设置字母间距为正常。

 

line-height: 1;：设置行高为 1。

 

margin: 0;：移除外边距。

 

overflow: visible;：设置内容溢出时可见。

 

padding: 0;：移除内边距。

 

text-align: left;：将文本左对齐。

 

text-decoration: none;：移除文本装饰（如下划线）。

 

text-indent: 0;：移除文本首行缩进。

 

text-shadow: none;：移除文本阴影。

 

text-transform: none;：移除文本转换（如大写或小写）。

 

visibility: visible;：设置元素可见性为可见。

 

white-space: normal;：设置空白符处理为正常，不进行额外处理。

 

word-spacing: normal;：设置单词间距为正常。
